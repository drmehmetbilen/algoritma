# Yeni bir kütüphane kurmak/silmek
pip install [kütüphaneAdı]
pip uninstall [kütüphaneAdı]

# Kütüphane yönetimi için kullanılan kodlar
pip list, pip show [kütühaneAdı]
pip freeze > requirements.txt

# Yeni bir python sanal çalışma ortamı oluşturmak içn
python -m venv .venv
# Mac'de aktive etmek
source .venv/bin/activate
# Windows'da aktive etmek
 .venv/scripts/activate
